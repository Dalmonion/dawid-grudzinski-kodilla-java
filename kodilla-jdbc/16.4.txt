SELECT I.SUMMARY, I.DESCRIPTION, I.USER_ID_ASSIGNEDTO AS ISSUESLIST
FROM ISSUES I JOIN ISSUESLISTS IL ON I.ISSUESLIST_ID = IL.ID;

SELECT I.SUMMARY, I.DESCRIPTION, U.FIRSTNAME AS ASSIGNED_USER_FIRSTNAME, U.LASTNAME AS ASSIGNED_USER_LASTNAME
FROM ISSUES I JOIN USERS U ON I.USER_ID_ASSIGNEDTO = U.ID;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS TASKS_ASSIGNED
FROM ISSUES I
JOIN USERS U ON I.USER_ID_ASSIGNEDTO = U.ID
GROUP BY I.USER_ID_ASSIGNEDTO;
